package com.integration.em.client360.exception;

import java.util.Map;

public class PipelineExceptionRecord {

    private String stage;
    private String sourceName;
    private int rowNumber;
    private String type;
    private String message;
    private String rawRow;

    public static PipelineExceptionRecord missing(
            String stage,
            String sourceName,
            int rowNumber,
            String field,
            Map<String, String> row
    ) {
        PipelineExceptionRecord r = new PipelineExceptionRecord();
        r.stage = stage;
        r.sourceName = sourceName;
        r.rowNumber = rowNumber;
        r.type = "MISSING_FIELD";
        r.message = "Missing required field: " + field;
        r.rawRow = row.toString();
        return r;
    }

    public String getStage() { return stage; }
    public String getSourceName() { return sourceName; }
    public int getRowNumber() { return rowNumber; }
    public String getType() { return type; }
    public String getMessage() { return message; }
    public String getRawRow() { return rawRow; }
}
