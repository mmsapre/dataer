package com.integration.em.client360.semantic;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;

import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.util.ArrayList;
import java.util.List;

public class LlmEmbeddingClient {

    private final HttpClient client = HttpClient.newHttpClient();
    private final ObjectMapper mapper = new ObjectMapper();

    private final String url;
    private final String token;
    private final String model;

    public LlmEmbeddingClient(String url, String token, String model) {
        this.url = url;
        this.token = token;
        this.model = model;
    }

    public List<Double> embed(String text) throws Exception {
        if (token == null || token.isBlank()) return List.of();

        String body = """
                {
                  "model": "%s",
                  "input": "%s"
                }
                """.formatted(escape(model), escape(text));

        HttpRequest req = HttpRequest.newBuilder()
                .uri(URI.create(url))
                .header("Authorization", "Bearer " + token)
                .header("Content-Type", "application/json")
                .POST(HttpRequest.BodyPublishers.ofString(body))
                .build();

        String response = client.send(req, HttpResponse.BodyHandlers.ofString()).body();

        JsonNode embedding = mapper.readTree(response)
                .path("data")
                .get(0)
                .path("embedding");

        List<Double> values = new ArrayList<>();
        for (JsonNode n : embedding) values.add(n.asDouble());

        return values;
    }

    private String escape(String v) {
        return v == null ? "" : v.replace("\"", "\\\"");
    }
}
